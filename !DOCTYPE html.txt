let currentUser=null,current=0,foundClues=[[],[],[]],totalCluesFound=0,startTime=null;

function show(id){document.querySelectorAll('.screen').forEach(s=>s.classList.remove('active'));document.getElementById(id).classList.add('active')}
function toast(msg,d=2500){const t=document.getElementById('toast');t.textContent=msg;t.classList.add('show');setTimeout(()=>t.classList.remove('show'),d)}
function hideLoading(){const el=document.getElementById('loading-overlay');el.style.opacity='0';setTimeout(()=>el.style.display='none',400)}
function updateProgress(n){document.getElementById('prog-fill').style.width=(n/chapters.length*100)+'%';document.getElementById('prog-label').textContent=`${n}/${chapters.length}`}

async function init(){
  const {data:{session}}=await sb.auth.getSession();
  if(session){currentUser=session.user;onLoggedIn();}
  else{show('screen-auth');}
  hideLoading();
}

function onLoggedIn(){
  document.getElementById('user-email-display').textContent=currentUser.email;
  document.getElementById('user-info').style.display='flex';
  document.getElementById('progress-area').style.display='flex';
  loadProgress();
}

async function doLogin(){
  const email=document.getElementById('login-email').value.trim();
  const pw=document.getElementById('login-password').value;
  const btn=document.getElementById('login-btn');
  if(!email||!pw){showAuthMsg('login-msg','请填写邮箱和密码','err');return}
  btn.disabled=true;btn.textContent='登录中…';
  const {data,error}=await sb.auth.signInWithPassword({email,password:pw});
  btn.disabled=false;btn.textContent='登录';
  if(error){showAuthMsg('login-msg','登录失败：'+error.message,'err');return}
  currentUser=data.user;onLoggedIn();show('screen-intro');
}

async function doRegister(){
  const email=document.getElementById('reg-email').value.trim();
  const pw=document.getElementById('reg-password').value;
  const btn=document.getElementById('reg-btn');
  if(!email||!pw){showAuthMsg('reg-msg','请填写邮箱和密码','err');return}
  if(pw.length<6){showAuthMsg('reg-msg','密码至少需要6位','err');return}
  btn.disabled=true;btn.textContent='注册中…';
  const {data,error}=await sb.auth.signUp({email,password:pw});
  btn.disabled=false;btn.textContent='注册并开始游戏';
  if(error){showAuthMsg('reg-msg','注册失败：'+error.message,'err');return}
  if(data.user&&!data.session){showAuthMsg('reg-msg','注册成功！请查收确认邮件后登录。','ok');return}
  currentUser=data.user;onLoggedIn();show('screen-intro');
}

async function signOut(){
  await sb.auth.signOut();currentUser=null;current=0;foundClues=[[],[],[]];totalCluesFound=0;
  document.getElementById('user-info').style.display='none';
  document.getElementById('progress-area').style.display='none';
  show('screen-auth');toast('已退出登录');
}

function showAuthMsg(id,msg,type){const el=document.getElementById(id);el.textContent=msg;el.className='auth-msg '+type}
function switchToRegister(){document.getElementById('auth-login').style.display='none';document.getElementById('auth-register').style.display='block'}
function switchToLogin(){document.getElementById('auth-register').style.display='none';document.getElementById('auth-login').style.display='block'}

async function loadProgress(){
  if(!currentUser)return;
  const {data}=await sb.from('game_progress').select('*').eq('user_id',currentUser.id).maybeSingle();
  if(data&&!data.completed){
    current=data.current_chapter||0;
    foundClues=data.found_clues||[[],[],[]];
    totalCluesFound=data.total_clues_found||0;
    startTime=data.start_time?new Date(data.start_time).getTime():Date.now();
    updateProgress(current);
    for(let i=0;i<current;i++){
      const el=document.getElementById('ch-preview-'+i);
      if(el){el.classList.add('done');el.querySelector('.ch-lock').textContent='✓ 已通关'}
    }
    if(current>0){document.getElementById('resume-note').style.display='block';document.getElementById('btn-start').textContent='继续挑战 →'}
  }
  show('screen-intro');
}

async function saveProgress(completed=false){
  if(!currentUser)return;
  const {error}=await sb.from('game_progress').upsert({
    user_id:currentUser.id,current_chapter:current,found_clues:foundClues,
    total_clues_found:totalCluesFound,
    start_time:startTime?new Date(startTime).toISOString():new Date().toISOString(),
    completed,updated_at:new Date().toISOString()
  },{onConflict:'user_id'});
  if(!error){const n=document.getElementById('saving-note');if(n){n.textContent='✓ 已保存';setTimeout(()=>{n.textContent=''},2000)}}
}

function startGame(){if(!startTime)startTime=Date.now();renderChapter();show('screen-game')}

function renderChapter(){
  const ch=chapters[current];
  document.getElementById('ch-tag').textContent=`关卡 ${String(current+1).padStart(2,'0')} / ${String(chapters.length).padStart(2,'0')}`;
  document.getElementById('ch-title').textContent=ch.title;
  document.getElementById('ch-desc').textContent=ch.desc;
  document.getElementById('clue-detail').textContent='点击上方按钮，检查现场……';
  document.getElementById('clue-detail').classList.remove('lit');
  document.getElementById('code-input').value='';
  document.getElementById('code-input').maxLength=ch.answer.length;
  document.getElementById('code-input').placeholder='?'.repeat(ch.answer.length);
  document.getElementById('feedback').textContent='';document.getElementById('feedback').className='feedback';
  document.getElementById('hint-box').textContent=ch.hint;document.getElementById('hint-box').style.display='none';
  renderClueGrid();
}

function renderClueGrid(){
  const ch=chapters[current];
  document.getElementById('clue-grid').innerHTML=ch.clues.map((c,i)=>
    `<button class="clue-btn${(foundClues[current]||[]).includes(i)?' found':''}" onclick="showClue(${i})">${c.label}</button>`
  ).join('');
}

function showClue(i){
  const ch=chapters[current];
  const box=document.getElementById('clue-detail');
  box.textContent=ch.clues[i].text;box.classList.add('lit');
  if(!(foundClues[current]||[]).includes(i)){
    if(!foundClues[current])foundClues[current]=[];
    foundClues[current].push(i);totalCluesFound++;
    renderClueGrid();saveProgress();
  }
}

function tryUnlock(){
  const val=document.getElementById('code-input').value.trim().toUpperCase();
  const ch=chapters[current];
  const fb=document.getElementById('feedback');
  if(val===ch.answer.toUpperCase()){
    fb.className='feedback ok';fb.textContent='√ 正确';
    current++;updateProgress(current);
    setTimeout(async()=>{
      if(current<chapters.length){document.getElementById('success-msg').textContent=ch.success;await saveProgress(false);show('screen-success')}
      else{await saveProgress(true);showWin()}
    },700);
  }else{
    fb.className='feedback err';fb.textContent='× 密码有误';
    setTimeout(()=>{fb.textContent='';fb.className='feedback'},2200);
  }
}

function nextChapter(){renderChapter();show('screen-game')}

function showWin(){
  const mins=startTime?Math.max(1,Math.round((Date.now()-startTime)/60000)):1;
  document.getElementById('stat-clues').textContent=totalCluesFound;
  document.getElementById('stat-time').textContent=mins;
  show('screen-win');toast('通关！进度已保存',3000);
}

async function restart(){
  current=0;foundClues=[[],[],[]];totalCluesFound=0;startTime=Date.now();
  await saveProgress(false);
  for(let i=0;i<chapters.length;i++){
    const el=document.getElementById('ch-preview-'+i);
    if(el){el.classList.remove('done');el.querySelector('.ch-lock').textContent=['4位数字','4位字母','1位数字'][i]}
  }
  document.getElementById('resume-note').style.display='none';
  document.getElementById('btn-start').textContent='开始挑战 →';
  updateProgress(0);show('screen-intro');
}
